CREATE TABLE ACCOUNT 
(ID VARCHAR2(40) NOT NULL PRIMARY KEY, 
PASSWORD VARCHAR2(40) NOT NULL, 
NAME VARCHAR2(40) 
NOT NULL)

CREATE TABLE POST
(NO NUMBER(6) NOT NULL PRIMARY KEY,
ACCOUNT_ID VARCHAR2(40) REFERENCES ACCOUNT(ID),
TYPE VARCHAR2(40) CHECK(TYPE IN('BUY', 'SELL')) NOT NULL,
TITLE VARCHAR2(4000) NOT NULL, 
PRICE NUMBER(12) NOT NULL, 
NOTE VARCHAR2(4000)  NOT NULL, 
REGIST DATE NOT NULL)
